Robust S3 → Hugo Sync (Production Grade)"



on:

  push:

    branches: [main]

  schedule:

    - cron: '0 0 * * *'

  workflow_dispatch:

    inputs:

      year:

        description: "Welk jaartal moet gesynct worden?"

        required: false

        default: "2024"



env:

  DEFAULT_YEAR: "2024"

  SOURCE_BUCKET: "publicaties-coevorden"

  DEST_BUCKET_PREFIX: "publicatie"

  AWS_REGION: "eu-central-1"



jobs:

  deploy:

    runs-on: ubuntu-latest

    permissions:

      contents: write



    steps:

      - name: Checkout repository

        uses: actions/checkout@v4



      - name: Setup Context

        run: |

          # Bepaal het jaar: Input -> Env -> Default

          SELECTED_YEAR="${{ github.event.inputs.year || env.DEFAULT_YEAR }}"

          echo "PROCESSING_YEAR=$SELECTED_YEAR" >> $GITHUB_ENV

          echo "TARGET_S3_BUCKET=${DEST_BUCKET_PREFIX}${SELECTED_YEAR}" >> $GITHUB_ENV



      - name: Install Tools

        run: |

          sudo apt-get update

          sudo apt-get install -y poppler-utils jq awscli



      - name: Sync S3 to Local & Convert

        env:

          AWS_ACCESS_KEY_ID: ${{ secrets.AWS_ACCESS_KEY_ID }}

          AWS_SECRET_ACCESS_KEY: ${{ secrets.AWS_SECRET_ACCESS_KEY }}

        run: |

          set -euo pipefail

          

          # 1. Maak mappen schoon

          mkdir -p ./s3_raw

          mkdir -p ./content/docs

          

          # 2. Sync ALLE PDF's in één keer (veel sneller dan losse cp's)

          echo "📥 Downloading PDFs for $PROCESSING_YEAR..."

          aws s3 sync "s3://$SOURCE_BUCKET/$PROCESSING_YEAR/" "./s3_raw/$PROCESSING_YEAR/" --quiet --exclude "*" --include "*.pdf"



          # 3. Genereer metadata lijst (Key + LastModified) via S3 API

          echo "📊 Fetching metadata..."

          aws s3api list-objects-v2 \

            --bucket "$SOURCE_BUCKET" \

            --prefix "$PROCESSING_YEAR/" \

            --query 'Contents[?ends_with(Key, `.pdf`)].[Key, LastModified]' \

            --output json > metadata.json



          # 4. Verwerkingsfunctie (nu zonder extra AWS calls!)

          process_pdf() {

            local key="$1"

            local mdate="$2"

            local clean_name=$(basename "${key%.pdf}")

            local target_md="content/docs/${key%.pdf}.md"

            

            mkdir -p "$(dirname "$target_md")"

            

            # Taxonomie opbouwen vanuit folderstructuur

            IFS='/' read -ra ADDR <<< "${key%/*}"

            local taxo_json=$(printf '%s\n' "${ADDR[@]}" | jq -R . | jq -s -c .)



            # Markdown schrijven

            cat <<EOF > "$target_md"

---

title: "${clean_name//\"/\\\"}"

date: "${mdate}"

maps: ${taxo_json}

source: "${key}"

---



EOF

            # Tekst extractie (foutbestendig)

            pdftotext -raw "./s3_raw/$key" - | tr -d '\000' >> "$target_md" || echo "Warning: Text extraction failed for $key"

          }



          export -f process_pdf

          

          # 5. Parallel verwerken met JQ en XARGS

          jq -r '.[] | @tsv' metadata.json | xargs -P 4 -L 1 bash -c 'process_pdf "$0" "$1"'

          

      - name: Install Hugo

        uses: peaceiris/actions-hugo@v3 # v3 is stabieler

        with:

          hugo-version: 'latest'

          extended: true



      - name: Build Hugo site

        run: |

          hugo --minify

          # Check of build geslaagd is

          [ -d "public" ] || exit 1



      - name: Index with Pagefind

        run: npx pagefind --site public --bundle-dir _pagefind



      - name: Deploy to S3

        env:

          AWS_ACCESS_KEY_ID: ${{ secrets.AWS_ACCESS_KEY_ID }}

          AWS_SECRET_ACCESS_KEY: ${{ secrets.AWS_SECRET_ACCESS_KEY }}

        run: |

          aws s3 sync public/ "s3://$TARGET_S3_BUCKET" \

            --delete \

            --region $AWS_REGION \

            --cache-control "max-age=31536000, public"
